mtk-msgs HTML Works

Copy and paste these examples to use MTKMsgs in your application

Info Message with Buttons
JavaScript Code
MTKMsgs.show({ type: 'info', icon: 'info', message: 'This is an informational message', buttons: [ { label: 'Learn More', action: 'learnMore' } ], closable: true, timer: null });
Warning Message
JavaScript Code
MTKMsgs.show({ type: 'warning', icon: 'warning', message: 'Warning: Please review your settings', buttons: [ { label: 'Review', action: 'review' }, { label: 'Dismiss', action: 'dismiss' } ], closable: true, timer: null });
Error Message (Screen Lock)
JavaScript Code
MTKMsgs.show({ type: 'error', icon: 'error', message: 'Error: Something went wrong', buttons: [ { label: 'Retry', action: 'retry' } ], closable: true, timer: null });
Success with 10s Timer
JavaScript Code
MTKMsgs.show({ type: 'success', icon: 'check_circle', message: 'Success: Operation completed', buttons: [], closable: false, timer: 10 });
Auto Info with 5s Timer
JavaScript Code
MTKMsgs.show({ type: 'info', icon: 'info', message: 'This message will close automatically', buttons: [], closable: false, timer: 5 });
Using Config Key
JavaScript Code
// Uses predefined config from mtk-msgs.config.js MTKMsgs.show('info');
Using wc.publish Event
JavaScript Code
wc.publish('mtk-msgs:show', { type: 'warning', icon: 'warning', message: 'Session will expire in 5 minutes', buttons: [ { label: 'Extend', action: 'extend' } ], closable: true, timer: null });
Hide Message
JavaScript Code
// Hide any currently displayed message MTKMsgs.hide();
Custom Error Message
JavaScript Code
MTKMsgs.show({ type: 'error', icon: 'error', message: 'Failed to save changes. Please try again.', buttons: [ { label: 'Retry', action: 'retry' }, { label: 'Cancel', action: 'cancel' } ], closable: true, timer: null });
Quick Success (3s Timer)
JavaScript Code
MTKMsgs.show({ type: 'success', icon: 'check_circle', message: 'File uploaded successfully!', buttons: [], closable: false, timer: 3 });
Warning with Timer + Close Button
JavaScript Code
MTKMsgs.show({ type: 'warning', icon: 'warning', message: 'Low disk space detected', buttons: [ { label: 'Clean Up', action: 'cleanup' } ], closable: true, timer: 15 });
Info without Close Button
JavaScript Code
MTKMsgs.show({ type: 'info', icon: 'info', message: 'Processing your request...', buttons: [], closable: false, timer: null });
Block Flag - Screen Locked While Shown
JavaScript Code
MTKMsgs.show({ type: 'warning', icon: 'warning', message: 'Screen is blocked while this message shows', buttons: [ { label: 'Dismiss', action: 'dismiss' } ], closable: true, timer: null, block: true, // Lock screen block2: false // Unlock when dismissed });
Block2 Flag - Overrides Block
JavaScript Code
MTKMsgs.show({ type: 'error', icon: 'error', message: 'Processing... Screen stays blocked after close', buttons: [ { label: 'Close Message', action: 'close' } ], closable: true, timer: null, block: false, // This is ignored block2: true // This overrides block flag }); // To unlock screen later, call: MTKMsgs.unblock();
